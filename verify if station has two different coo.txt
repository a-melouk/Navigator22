fetch(baseURI + 'lines/tramway/stations')
  .then((response) => response.json())
  .then((data) => {
    console.log('Tramway')
    for (let i = 0; i < data.length - 1; i++) {
      if (data[i].to.name === data[i + 1].from.name) {
        if (trueIfDifferent(data[i].to.coordinates, data[i + 1].from.coordinates)) {
          console.log(data[i].to.name, data[i + 1].from.name)
        }
      }
    }
    console.log('')
  })
